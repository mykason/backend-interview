package leet.hot100;

public class AddTwoNumbers_2_H {
    public static class ListNode {
        int val;
        ListNode next;
        ListNode() {}
        ListNode(int val) { this.val = val; }
        ListNode(int val, ListNode next) { this.val = val; this.next = next; }
    }

    /**
     * 【两数相加】
     * 给你两个非空链表，表示两个非负整数，数字按逆序存储。请将两数相加，并以相同形式返回链表。
     * 示例：输入 l1=[2,4,3], l2=[5,6,4]，输出 [7,0,8]。
     */
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode dummy = new ListNode(0), cur = dummy;
        int carry = 0;
        while (l1 != null || l2 != null || carry != 0) {
            int sum = carry;
            if (l1 != null) { sum += l1.val; l1 = l1.next; }
            if (l2 != null) { sum += l2.val; l2 = l2.next; }
            // 当前位保存个位，十位作为进位
            cur.next = new ListNode(sum % 10);
            carry = sum / 10;
            cur = cur.next;
        }
        return dummy.next;
    }

    /**
     * 解题技巧/思路
     * - 按链表顺序逐位相加，维护 carry。
     * - 虚拟头节点简化结果链表拼接。
     * - 时间复杂度 O(max(m,n))，空间复杂度 O(max(m,n))。
     */
}

package leet.hot100;

import java.util.HashMap;
import java.util.Map;

public class BuildTree_105_H {
    public static class TreeNode { int val; TreeNode left; TreeNode right; TreeNode() {} TreeNode(int val) { this.val = val; } TreeNode(int val, TreeNode left, TreeNode right) { this.val = val; this.left = left; this.right = right; } }
    private Map<Integer, Integer> index;

    /**
     * 【从前序与中序遍历序列构造二叉树】
     * 给定 preorder 和 inorder，构造并返回二叉树。
     * 示例：输入 preorder=[3,9,20,15,7], inorder=[9,3,15,20,7]，输出对应二叉树。
     */
    public TreeNode buildTree(int[] preorder, int[] inorder) {
        index = new HashMap<>();
        for (int i = 0; i < inorder.length; i++) index.put(inorder[i], i);
        return build(preorder, 0, preorder.length - 1, 0, inorder.length - 1);
    }

    private TreeNode build(int[] preorder, int ps, int pe, int is, int ie) {
        if (ps > pe) return null;
        TreeNode root = new TreeNode(preorder[ps]);
        int ri = index.get(root.val), leftSize = ri - is;
        root.left = build(preorder, ps + 1, ps + leftSize, is, ri - 1);
        root.right = build(preorder, ps + leftSize + 1, pe, ri + 1, ie);
        return root;
    }

    /** 解题技巧/思路
     * - 前序首元素是根，中序根位置划分左右子树。
     * - 哈希表加速中序下标查询。
     * - 时间复杂度 O(n)，空间复杂度 O(n)。
     */
}

package leet.hot100;

import java.util.ArrayList;
import java.util.LinkedList;
import java.util.List;
import java.util.Queue;

public class CanFinish_207_H {
    /**
     * 【课程表】
     * 给你课程数 numCourses 和先修关系 prerequisites，判断是否能完成所有课程。
     * 示例：输入 numCourses=2, prerequisites=[[1,0]]，输出 true。
     */
    public boolean canFinish(int numCourses, int[][] prerequisites) {
        List<List<Integer>> graph = new ArrayList<>();
        for (int i = 0; i < numCourses; i++) graph.add(new ArrayList<>());
        int[] indeg = new int[numCourses];
        for (int[] p : prerequisites) { graph.get(p[1]).add(p[0]); indeg[p[0]]++; }
        Queue<Integer> q = new LinkedList<>();
        for (int i = 0; i < numCourses; i++) if (indeg[i] == 0) q.offer(i);
        int count = 0;
        while (!q.isEmpty()) {
            int cur = q.poll(); count++;
            for (int next : graph.get(cur)) if (--indeg[next] == 0) q.offer(next);
        }
        return count == numCourses;
    }

    /** 解题技巧/思路
     * - 拓扑排序检测有向图是否有环。
     * - 不断移除入度为 0 的课程。
     * - 时间复杂度 O(n+m)，空间复杂度 O(n+m)。
     */
}
```

    /**
     * 【跳跃游戏】
     * 给定非负整数数组 nums，初始在第一个下标，判断是否能到达最后一个下标。
     * 示例：输入 [2,3,1,1,4]，输出 true。

    public boolean canJump(int[] nums) {
        int far = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i > far) return false;
            far = Math.max(far, i + nums[i]);
        }
        return true;

     * - 若遍历位置超过最远位置，说明不可达。
     * - 时间复杂度 O(n)，空间复杂度 O(1)。
     */
}

package leet.hot100;

public class CanPartition_416_H {
    /**
     * 【分割等和子集】
     * 给你只包含正整数的数组 nums，判断是否可以分成两个和相等的子集。
     * 示例：输入 [1,5,11,5]，输出 true。
     */
    public boolean canPartition(int[] nums) {
        int sum = 0;
        for (int num : nums) sum += num;
        if ((sum & 1) == 1) return false;
        int target = sum / 2;
        boolean[] dp = new boolean[target + 1];
        dp[0] = true;
        for (int num : nums) {
            for (int j = target; j >= num; j--) dp[j] |= dp[j - num];
        }
        return dp[target];
    }

    /** 解题技巧/思路
     * - 问题转化为能否选出若干数和为总和一半。
     * - 0/1 背包倒序更新，避免重复使用元素。
     * - 时间复杂度 O(n*sum)，空间复杂度 O(sum)。
     */
}

package leet.hot100;

import java.util.Arrays;

public class CoinChange_322_H {
    /**
     * 【零钱兑换】
     * 给你硬币数组 coins 和金额 amount，返回凑成总金额所需的最少硬币数；不能凑成返回 -1。
     * 示例：输入 coins=[1,2,5], amount=11，输出 3。
     */
    public int coinChange(int[] coins, int amount) {
        int[] dp = new int[amount + 1];
        Arrays.fill(dp, amount + 1);
        dp[0] = 0;
        for (int coin : coins) {
            for (int i = coin; i <= amount; i++) dp[i] = Math.min(dp[i], dp[i - coin] + 1);
        }
        return dp[amount] > amount ? -1 : dp[amount];
    }

    /** 解题技巧/思路
     * - 完全背包：每种硬币可重复选择。
     * - dp[i] 表示凑成金额 i 的最少硬币数。
     * - 时间复杂度 O(amount*coins)，空间复杂度 O(amount)。
     */
}

package leet.hot100;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class CombinationSum_39_H {
    /**
     * 【组合总和】
     * 给你无重复整数数组 candidates 和目标数 target，找出所有和为 target 的组合，同一数字可重复使用。
     * 示例：输入 candidates=[2,3,6,7], target=7，输出 [[2,2,3],[7]]。
     */
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> ans = new ArrayList<>();
        backtrack(candidates, target, 0, new ArrayList<>(), ans);
        return ans;
    }

    private void backtrack(int[] candidates, int remain, int start, List<Integer> path, List<List<Integer>> ans) {
        if (remain == 0) { ans.add(new ArrayList<>(path)); return; }
        for (int i = start; i < candidates.length && candidates[i] <= remain; i++) {
            path.add(candidates[i]);
            backtrack(candidates, remain - candidates[i], i, path, ans);
            path.remove(path.size() - 1);
        }
    }

    /** 解题技巧/思路
     * - 回溯枚举选择，start 控制组合顺序避免重复。
     * - 同一数字可重复使用，递归时仍传入 i。
     * - 时间复杂度与答案数量相关，空间复杂度 O(target/min)。
     */
}
```

import java.util.HashMap;
import java.util.Map;

public class CopyRandomList_138_H {
    static class Node { int val; Node next; Node random; Node(int val) { this.val = val; } }

    /**
     * 【随机链表的复制】
     * 给你一个长度为 n 的链表，每个节点包含 next 和 random 指针，返回该链表的深拷贝。
     * 示例：输入 head=[[7,null],[13,0],[11,4],[10,2],[1,0]]，输出深拷贝链表。
     */
    public Node copyRandomList(Node head) {
        if (head == null) return null;
        Map<Node, Node> map = new HashMap<>();
        for (Node cur = head; cur != null; cur = cur.next) map.put(cur, new Node(cur.val));
        for (Node cur = head; cur != null; cur = cur.next) {
            map.get(cur).next = map.get(cur.next);
            map.get(cur).random = map.get(cur.random);
        }
        return map.get(head);
    }

    /** 解题技巧/思路
     * - 哈希表建立原节点到新节点的映射。
     * - 第二次遍历补齐 next 和 random 指针。
     * - 时间复杂度 O(n)，空间复杂度 O(n)。
     */
}

package leet.hot100;

public class CountSubstrings_647_H {
    /**
     * 【回文子串】
     * 给你字符串 s，返回其中回文子串的数目。
     * 示例：输入 s="aaa"，输出 6。
     */
    public int countSubstrings(String s) {
        int ans = 0;
        for (int i = 0; i < s.length(); i++) {
            ans += expand(s, i, i);
            ans += expand(s, i, i + 1);
        }
        return ans;
    }

    private int expand(String s, int l, int r) {
        int count = 0;
        while (l >= 0 && r < s.length() && s.charAt(l--) == s.charAt(r++)) count++;
        return count;
    }

    /** 解题技巧/思路
     * - 枚举奇数和偶数回文中心。
     * - 每向外扩展成功一次，就得到一个回文子串。
     * - 时间复杂度 O(n^2)，空间复杂度 O(1)。
     */
}

package leet.hot100;

import java.util.ArrayDeque;
import java.util.Deque;

public class DailyTemperatures_739_H {
    /**
     * 【每日温度】
     * 给定整数数组 temperatures，返回每一天需要等几天才会有更高温度；没有则为 0。
     * 示例：输入 [73,74,75,71,69,72,76,73]，输出 [1,1,4,2,1,1,0,0]。
     */
    public int[] dailyTemperatures(int[] temperatures) {
        int[] ans = new int[temperatures.length];
        Deque<Integer> stack = new ArrayDeque<>();
        for (int i = 0; i < temperatures.length; i++) {
            while (!stack.isEmpty() && temperatures[i] > temperatures[stack.peek()]) {
                int idx = stack.pop();
                ans[idx] = i - idx;
            }
            stack.push(i);
        }
        return ans;
    }

    /** 解题技巧/思路
     * - 单调栈保存尚未找到更高温度的下标。
     * - 当前温度更高时，弹出并计算等待天数。
     * - 时间复杂度 O(n)，空间复杂度 O(n)。
     */
}

package leet.hot100;

import java.util.ArrayDeque;
import java.util.Deque;

public class DecodeString_394_H {
    /**
     * 【字符串解码】
     * 给定编码字符串，规则为 k[encoded_string]，返回解码后的字符串。
     * 示例：输入 s="3[a]2[bc]"，输出 "aaabcbc"。
     */
    public String decodeString(String s) {
        Deque<Integer> nums = new ArrayDeque<>();
        Deque<StringBuilder> strs = new ArrayDeque<>();
        StringBuilder cur = new StringBuilder();
        int num = 0;
        for (char c : s.toCharArray()) {
            if (Character.isDigit(c)) num = num * 10 + c - '0';
            else if (c == '[') { nums.push(num); strs.push(cur); num = 0; cur = new StringBuilder(); }
            else if (c == ']') {
                StringBuilder prev = strs.pop();
                int times = nums.pop();
                while (times-- > 0) prev.append(cur);
                cur = prev;
            } else cur.append(c);
        }
        return cur.toString();
    }

    /** 解题技巧/思路
     * - 遇到 [ 保存重复次数和之前字符串。
     * - 遇到 ] 弹出并展开当前片段。
     * - 时间复杂度 O(解码后长度)，空间复杂度 O(嵌套深度+结果长度)。
     */
}

package leet.hot100;

public class DetectCycle_142_H {
    public static class ListNode { int val; ListNode next; ListNode(int x) { val = x; } }

    /**
     * 【环形链表 II】
     * 给定链表头节点 head，返回链表开始入环的第一个节点；无环返回 null。
     * 示例：输入 head=[3,2,0,-4], pos=1，输出索引 1 的节点。
     */
    public ListNode detectCycle(ListNode head) {
        ListNode slow = head, fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next; fast = fast.next.next;
            if (slow == fast) {
                ListNode p = head;
                while (p != slow) { p = p.next; slow = slow.next; }
                return p;
            }
        }
        return null;
    }

    /** 解题技巧/思路
     * - 快慢指针先判断是否有环。
     * - 相遇后从头和相遇点同步走，相遇处就是入环点。
     * - 时间复杂度 O(n)，空间复杂度 O(1)。
     */
}

package leet.hot100;

public class DiameterOfBinaryTree_543_H {
    public static class TreeNode { int val; TreeNode left; TreeNode right; TreeNode() {} TreeNode(int val) { this.val = val; } TreeNode(int val, TreeNode left, TreeNode right) { this.val = val; this.left = left; this.right = right; } }
    private int ans;

    /**
     * 【二叉树的直径】
     * 给你二叉树根节点 root，返回任意两个节点之间最长路径的边数。
     * 示例：输入 [1,2,3,4,5]，输出 3。
     */
    public int diameterOfBinaryTree(TreeNode root) {
        depth(root);
        return ans;
    }

    private int depth(TreeNode node) {
        if (node == null) return 0;
        int l = depth(node.left), r = depth(node.right);
        ans = Math.max(ans, l + r);
        return Math.max(l, r) + 1;
    }

    /** 解题技巧/思路
     * - 经过某节点的最长路径为左深度加右深度。
     * - 后序遍历计算深度并更新答案。
     * - 时间复杂度 O(n)，空间复杂度 O(h)。
     */
}

package leet.hot100;

public class Exist_79_H {
    /**
     * 【单词搜索】
     * 给定字符网格 board 和字符串 word，判断 word 是否存在于网格中，相邻单元格可水平或垂直连接。
     * 示例：输入 board 和 word="ABCCED"，输出 true。
     */
    public boolean exist(char[][] board, String word) {
        for (int i = 0; i < board.length; i++)
            for (int j = 0; j < board[0].length; j++)
                if (dfs(board, word, i, j, 0)) return true;
        return false;
    }

    private boolean dfs(char[][] board, String word, int i, int j, int k) {
        if (k == word.length()) return true;
        if (i < 0 || i == board.length || j < 0 || j == board[0].length || board[i][j] != word.charAt(k)) return false;
        char old = board[i][j];
        board[i][j] = '#';
        boolean ok = dfs(board, word, i + 1, j, k + 1) || dfs(board, word, i - 1, j, k + 1) || dfs(board, word, i, j + 1, k + 1) || dfs(board, word, i, j - 1, k + 1);
        board[i][j] = old;
        return ok;
    }

    /** 解题技巧/思路
     * - 从每个格子出发 DFS 匹配单词。
     * - 临时标记已访问格子，回溯时恢复。
     * - 时间复杂度 O(mn*3^L)，空间复杂度 O(L)。
     */
}

package leet.hot100;

import java.util.ArrayList;
import java.util.List;

public class FindAnagrams_438_H {
    /**
     * 【找到字符串中所有字母异位词】
     * 给定字符串 s 和 p，找到 s 中所有 p 的异位词子串起始索引。
     * 示例：输入 s="cbaebabacd", p="abc"，输出 [0,6]。
     */
    public List<Integer> findAnagrams(String s, String p) {
        List<Integer> ans = new ArrayList<>();
        if (s.length() < p.length()) return ans;
        int[] cnt = new int[26];
        for (char c : p.toCharArray()) cnt[c - 'a']++;
        int left = 0, right = 0, need = p.length();
        while (right < s.length()) {
            if (cnt[s.charAt(right++) - 'a']-- > 0) need--;
            if (right - left > p.length() && ++cnt[s.charAt(left++) - 'a'] > 0) need++;
            if (need == 0) ans.add(left);
        }
        return ans;
    }

    /** 解题技巧/思路
     * - 固定长度滑动窗口维护字符差值。
     * - need 为还需要匹配的字符数量，等于 0 时窗口合法。
     * - 时间复杂度 O(n)，空间复杂度 O(1)。
     */
}

package leet.hot100;

import java.util.ArrayList;
import java.util.List;

public class FindDisappearedNumbers_448_H {
    /**
     * 【找到所有数组中消失的数字】
     * 给你含 n 个整数的数组 nums，其中 nums[i] 在 [1,n]，返回范围内没有出现的数字。
     * 示例：输入 [4,3,2,7,8,2,3,1]，输出 [5,6]。

    public List<Integer> findDisappearedNumbers(int[] nums) {

            int idx = Math.abs(num) - 1;

package leet.hot100;

import java.util.PriorityQueue;

public class FindKthLargest_215_H {
    /**
     * 【数组中的第 K 个最大元素】
     * 给定整数数组 nums 和整数 k，返回数组中第 k 个最大的元素。
     * 示例：输入 nums=[3,2,1,5,6,4], k=2，输出 5。
     */
    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> heap = new PriorityQueue<>();
        for (int num : nums) {
            heap.offer(num);
            if (heap.size() > k) heap.poll();
        }
        return heap.peek();
    }

    /** 解题技巧/思路
     * - 维护容量为 k 的小顶堆。
     * - 堆顶就是当前前 k 大中的最小值，即第 k 大。
     * - 时间复杂度 O(nlogk)，空间复杂度 O(k)。
     */

     * 【目标和】
     * 给你整数数组 nums 和整数 target，给每个数添加 + 或 -，返回表达式结果为 target 的方法数。
     * 示例：输入 nums=[1,1,1,1,1], target=3，输出 5。

        for (int num : nums) sum += num;

public class LRUCache_146_H {
    /**
     * 【LRU 缓存】
     * 请设计并实现一个满足 LRU 最近最少使用缓存约束的数据结构。
     * get 和 put 必须以 O(1) 平均时间复杂度运行。
     */
    private static class Node { int key, val; Node prev, next; Node(int key, int val) { this.key = key; this.val = val; } }
    private final int capacity;
    private final Map<Integer, Node> map = new HashMap<>();
    private final Node head = new Node(0, 0), tail = new Node(0, 0);

    public LRUCache_146_H(int capacity) {
        this.capacity = capacity;
        head.next = tail; tail.prev = head;
    }

    public int get(int key) {
        Node node = map.get(key);
        if (node == null) return -1;
        moveToHead(node);
        return node.val;
    }

    public void put(int key, int value) {
        Node node = map.get(key);
        if (node != null) { node.val = value; moveToHead(node); return; }
        node = new Node(key, value);
        map.put(key, node); addFirst(node);
        if (map.size() > capacity) { Node last = tail.prev; remove(last); map.remove(last.key); }
    }

    private void moveToHead(Node node) { remove(node); addFirst(node); }
    private void addFirst(Node node) { node.next = head.next; node.prev = head; head.next.prev = node; head.next = node; }
    private void remove(Node node) { node.prev.next = node.next; node.next.prev = node.prev; }

    /** 解题技巧/思路
     * - 哈希表负责 O(1) 定位节点，双向链表维护最近使用顺序。
     * - 访问或更新后移动到头部，容量超限删除尾部。

            ans = Math.max(ans, right - left + 1);
        }
        return ans;

     * 解题技巧/思路
     * - 滑动窗口维护无重复区间。
     * - 遇到重复字符时，左边界跳到上次出现位置之后。

     * 示例：输入 digits="23"，输出 ["ad","ae","af","bd","be","bf","cd","ce","cf"]。
     */
    public List<String> letterCombinations(String digits) {
        List<String> ans = new ArrayList<>();
        if (digits == null || digits.length() == 0) return ans;
        backtrack(digits, 0, new StringBuilder(), ans);
        return ans;
    }

    private void backtrack(String digits, int index, StringBuilder path, List<String> ans) {
        if (index == digits.length()) { ans.add(path.toString()); return; }
        String letters = MAP[digits.charAt(index) - '0'];
        for (int i = 0; i < letters.length(); i++) {
            path.append(letters.charAt(i));
            backtrack(digits, index + 1, path, ans);
            path.deleteCharAt(path.length() - 1);
        }
    }

    /** 解题技巧/思路
     * - 每一位数字对应一组选项，使用回溯枚举组合。

     * 给你二叉树根节点 root，返回其节点值的层序遍历。
     * 示例：输入 root=[3,9,20,null,null,15,7]，输出 [[3],[9,20],[15,7]]。
     */
    public List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> ans = new ArrayList<>();
        if (root == null) return ans;
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        while (!queue.isEmpty()) {
            int size = queue.size();
            List<Integer> level = new ArrayList<>();
            for (int i = 0; i < size; i++) {
                TreeNode node = queue.poll();
                level.add(node.val);
                if (node.left != null) queue.offer(node.left);
                if (node.right != null) queue.offer(node.right);
            }
            ans.add(level);
        }
        return ans;
    }

    /** 解题技巧/思路
     * - BFS 队列按层处理。
     * - 每轮固定当前层 size，避免和下一层混淆。
     * - 时间复杂度 O(n)，空间复杂度 O(n)。

public class LongestPalindrome_5_H {
    /**
     * 【最长回文子串】
     * 给你一个字符串 s，返回 s 中最长的回文子串。
     * 示例：输入 s="babad"，输出 "bab" 或 "aba"。
     */
    public String longestPalindrome(String s) {
        int start = 0, end = 0;
        for (int i = 0; i < s.length(); i++) {
            int len1 = expand(s, i, i);
            int len2 = expand(s, i, i + 1);
            int len = Math.max(len1, len2);
            if (len > end - start + 1) {
                start = i - (len - 1) / 2;
                end = i + len / 2;
            }
        }
        return s.substring(start, end + 1);
    }

    private int expand(String s, int left, int right) {
        while (left >= 0 && right < s.length() && s.charAt(left) == s.charAt(right)) {
            left--;
            right++;
        }
        return right - left - 1;
    }

    /**
     * 解题技巧/思路
     * - 枚举每个位置作为奇数/偶数回文中心并向两侧扩展。
     * - 每次记录更长的回文边界。

     *
     * 你可以假设数组是非空的，并且给定的数组总是存在多数元素。

     *
     * 提示：
     * n == nums.length
     * 1 <= n <= 5 * 104
     * -109 <= nums[i] <= 109
     */
    public int majorityElement(int[] nums) {
        int n = nums.length;
        // 统计每个元素出现的次数
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < n; i ++){
            if(map.containsKey(nums[i])){
                Integer val = map.get(nums[i]);
                map.put(nums[i], ++val);

                map.put(nums[i], 1);

     * 【下一个排列】
     * 将整数数组重新排列成字典序中下一个更大的排列；若不存在，则升序排列。
     * 示例：输入 [1,2,3]，输出 [1,3,2]。

        while (i >= 0 && nums[i] >= nums[i + 1]) i--;
        if (i >= 0) {
            int j = nums.length - 1;
            while (nums[j] <= nums[i]) j--;

import java.util.LinkedList;
import java.util.Queue;

public class OrangesRotting_994_H {
    /**
     * 【腐烂的橘子】
     * 给定网格，0 为空、1 为新鲜橘子、2 为腐烂橘子，每分钟腐烂相邻新鲜橘子，返回全部腐烂所需分钟数。
     * 示例：输入 [[2,1,1],[1,1,0],[0,1,1]]，输出 4。
     */
    public int orangesRotting(int[][] grid) {
        Queue<int[]> q = new LinkedList<>();
        int fresh = 0, minutes = 0;
        for (int i = 0; i < grid.length; i++) for (int j = 0; j < grid[0].length; j++) {
            if (grid[i][j] == 2) q.offer(new int[]{i, j});
            else if (grid[i][j] == 1) fresh++;
        }
        int[][] dirs = {{1,0},{-1,0},{0,1},{0,-1}};
        while (fresh > 0 && !q.isEmpty()) {
            int size = q.size(); minutes++;
            for (int s = 0; s < size; s++) {
                int[] cur = q.poll();
                for (int[] d : dirs) {
                    int x = cur[0] + d[0], y = cur[1] + d[1];
                    if (x >= 0 && x < grid.length && y >= 0 && y < grid[0].length && grid[x][y] == 1) {
                        grid[x][y] = 2; fresh--; q.offer(new int[]{x, y});
                    }
                }
            }
        }
        return fresh == 0 ? minutes : -1;
    }

    /** 解题技巧/思路
     * - 多源 BFS：所有腐烂橘子同时作为起点。
     * - 每一层 BFS 表示一分钟。
     * - 时间复杂度 O(mn)，空间复杂度 O(mn)。
     */

package leet.hot100;

import java.util.ArrayList;
import java.util.List;

public class Partition_131_H {
    /**
     * 【分割回文串】
     * 给你字符串 s，将 s 分割成若干子串，使每个子串都是回文串，返回所有方案。
     * 示例：输入 s="aab"，输出 [["a","a","b"],["aa","b"]]。
     */
    public List<List<String>> partition(String s) {
        List<List<String>> ans = new ArrayList<>();
        backtrack(s, 0, new ArrayList<>(), ans);
        return ans;
    }

    private void backtrack(String s, int start, List<String> path, List<List<String>> ans) {
        if (start == s.length()) { ans.add(new ArrayList<>(path)); return; }
        for (int end = start; end < s.length(); end++) {
            if (isPalindrome(s, start, end)) {
                path.add(s.substring(start, end + 1));
                backtrack(s, end + 1, path, ans);
                path.remove(path.size() - 1);
            }
        }
    }

    private boolean isPalindrome(String s, int l, int r) {
        while (l < r) if (s.charAt(l++) != s.charAt(r--)) return false;
        return true;
    }

    /** 解题技巧/思路
     * - 枚举每个切割位置，只有回文子串才能加入路径。
     * - 回溯遍历所有合法切割方案。
     * - 时间复杂度 O(n*2^n)，空间复杂度 O(n)。
     */
}

public class PathSum_437_H {
    public static class TreeNode { int val; TreeNode left; TreeNode right; TreeNode() {} TreeNode(int val) { this.val = val; } TreeNode(int val, TreeNode left, TreeNode right) { this.val = val; this.left = left; this.right = right; } }

    /**
     * 【路径总和 III】
     * 给定二叉树 root 和目标和 targetSum，返回路径和等于目标值的路径数。
     * 示例：输入 root=[10,5,-3,3,2,null,11,3,-2,null,1], targetSum=8，输出 3。
     */
    public int pathSum(TreeNode root, int targetSum) {
        Map<Long, Integer> map = new HashMap<>();
        map.put(0L, 1);
        return dfs(root, 0, targetSum, map);
    }

    private int dfs(TreeNode node, long sum, int target, Map<Long, Integer> map) {
        if (node == null) return 0;
        sum += node.val;
        int ans = map.getOrDefault(sum - target, 0);
        map.put(sum, map.getOrDefault(sum, 0) + 1);
        ans += dfs(node.left, sum, target, map) + dfs(node.right, sum, target, map);
        map.put(sum, map.get(sum) - 1);
        return ans;
    }

    /** 解题技巧/思路
     * - 前缀和记录根到当前路径上的累加和。
     * - 若存在前缀 sum-target，则对应一条合法路径。
     * - 时间复杂度 O(n)，空间复杂度 O(h)。
     */

import java.util.List;

public class Permute_46_H {
    /**
     * 【全排列】
     * 给定不含重复数字的数组 nums，返回所有可能的全排列。
     * 示例：输入 [1,2,3]，输出所有 6 个排列。
     */
    public List<List<Integer>> permute(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        backtrack(nums, new boolean[nums.length], new ArrayList<>(), ans);
        return ans;
    }

    private void backtrack(int[] nums, boolean[] used, List<Integer> path, List<List<Integer>> ans) {
        if (path.size() == nums.length) { ans.add(new ArrayList<>(path)); return; }
        for (int i = 0; i < nums.length; i++) {
            if (used[i]) continue;
            used[i] = true; path.add(nums[i]);

            path.remove(path.size() - 1); used[i] = false;
        }
    }

package leet.hot100;

public class SearchRange_34_H {
    /**
     * 【在排序数组中查找元素的第一个和最后一个位置】
     * 给定升序数组 nums 和目标值 target，返回起始和结束位置，不存在返回 [-1,-1]。
     * 示例：输入 nums=[5,7,7,8,8,10], target=8，输出 [3,4]。
     */
    public int[] searchRange(int[] nums, int target) {
        int left = lowerBound(nums, target);
        int right = lowerBound(nums, target + 1) - 1;
        if (left == nums.length || nums[left] != target) return new int[]{-1, -1};
        return new int[]{left, right};
    }

    private int lowerBound(int[] nums, int target) {
        int l = 0, r = nums.length;
        while (l < r) {
            int m = l + (r - l) / 2;
            if (nums[m] >= target) r = m; else l = m + 1;
        }
        return l;
    }

    /** 解题技巧/思路
     * - 两次二分分别找第一个 >=target 和第一个 >target 的位置。
     * - 右边界是第二个位置减一。

public class Search_33_H {
    /**
     * 【搜索旋转排序数组】
     * 整数数组按升序排列后在某点旋转，给定 target，返回下标，不存在返回 -1。
     * 示例：输入 nums=[4,5,6,7,0,1,2], target=0，输出 4。
     */
    public int search(int[] nums, int target) {
        int left = 0, right = nums.length - 1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target) return mid;
            if (nums[left] <= nums[mid]) {
                if (nums[left] <= target && target < nums[mid]) right = mid - 1;
                else left = mid + 1;
            } else {
                if (nums[mid] < target && target <= nums[right]) left = mid + 1;
                else right = mid - 1;
            }
        }
        return -1;
    }

    /** 解题技巧/思路
     * - 二分时总有一侧是有序区间。
     * - 判断 target 是否在有序侧，决定保留哪半边。
     * - 时间复杂度 O(log n)，空间复杂度 O(1)。
     */

package leet.hot100;

public class SortColors_75_H {
    /**
     * 【颜色分类】
     * 题目描述
     * 给定一个包含红色、白色和蓝色、共 n 个元素的数组 nums，原地对它们进行排序，使得相同颜色的元素相邻，并按照红色、白色、蓝色顺序排列。
     *
     * 我们使用整数 0、1 和 2 分别表示红色、白色和蓝色。
     *
     * 必须在不使用库内置的 sort 函数的情况下解决这个问题。
     *
     * 示例 1：
     * 输入：nums = [2,0,2,1,1,0]
     * 输出：[0,0,1,1,2,2]
     *
     * 示例 2：
     * 输入：nums = [2,0,1]
     * 输出：[0,1,2]
     *
     * 提示：
     * n == nums.length
     * 1 <= n <= 300
     * nums[i] 为 0、1 或 2
     *
     * 进阶：
     * 你能想出一个仅使用常数空间的一趟扫描算法吗？
     */
    public void sortColors(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        int i = 0;

        while (i <= right) {
            if (nums[i] == 0) {
                // 0 放到左侧已排序区域，当前位置继续向右扫描
                swap(nums, i, left);
                left++;
                i++;
            } else if (nums[i] == 2) {
                // 2 放到右侧已排序区域，换回来的数还未检查，i 不移动
                swap(nums, i, right);
                right--;
            } else {
                i++;
            }
        }
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

    /**
     * 解题技巧/思路
     * - 使用三指针维护三个区域：
     *   [0, left) 都是 0，(right, n - 1] 都是 2，i 负责扫描未知区域。
     * - 遇到 0：和 left 交换，left 与 i 同时右移，因为换过来的位置已经处理过。
     * - 遇到 2：和 right 交换，只右移 right，不移动 i，因为从右侧换回来的值仍需判断。
     * - 遇到 1：本身属于中间区域，直接跳过。
     * - 时间复杂度：O(n)，每个元素最多被检查和交换常数次。
     * - 空间复杂度：O(1)，只使用常数个指针变量。
     */
}

public class SortList_148_H {
    public static class ListNode { int val; ListNode next; ListNode() {} ListNode(int val) { this.val = val; } ListNode(int val, ListNode next) { this.val = val; this.next = next; } }

    /**
     * 【排序链表】
     * 给你链表头节点 head，请将其按升序排列并返回排序后的链表。
     * 示例：输入 head=[4,2,1,3]，输出 [1,2,3,4]。
     */
    public ListNode sortList(ListNode head) {
        if (head == null || head.next == null) return head;
        ListNode slow = head, fast = head.next;
        while (fast != null && fast.next != null) { slow = slow.next; fast = fast.next.next; }
        ListNode mid = slow.next; slow.next = null;
        return merge(sortList(head), sortList(mid));
    }

    private ListNode merge(ListNode a, ListNode b) {
        ListNode dummy = new ListNode(0), cur = dummy;
        while (a != null && b != null) {
            if (a.val <= b.val) { cur.next = a; a = a.next; } else { cur.next = b; b = b.next; }
            cur = cur.next;
        }
        cur.next = a != null ? a : b;
        return dummy.next;
    }

    /** 解题技巧/思路
     * - 链表适合归并排序：快慢指针拆分，递归排序后合并。
     * - 合并两个有序链表时复用原节点。
     * - 时间复杂度 O(nlogn)，空间复杂度 O(logn)。
     */
}

public class SpiralOrder_54_H {
    /**
     * 【螺旋矩阵】
     * 给你 m x n 矩阵 matrix，按照顺时针螺旋顺序返回所有元素。
     * 示例：输入 [[1,2,3],[4,5,6],[7,8,9]]，输出 [1,2,3,6,9,8,7,4,5]。
     */
    public List<Integer> spiralOrder(int[][] matrix) {
        List<Integer> ans = new ArrayList<>();
        int top = 0, bottom = matrix.length - 1, left = 0, right = matrix[0].length - 1;
        while (top <= bottom && left <= right) {
            for (int j = left; j <= right; j++) ans.add(matrix[top][j]);
            top++;
            for (int i = top; i <= bottom; i++) ans.add(matrix[i][right]);
            right--;
            if (top <= bottom) for (int j = right; j >= left; j--) ans.add(matrix[bottom][j]);
            bottom--;
            if (left <= right) for (int i = bottom; i >= top; i--) ans.add(matrix[i][left]);
            left++;
        }
        return ans;
    }

    /** 解题技巧/思路
     * - 用四个边界表示当前未遍历矩形。
     * - 每走完一条边就收缩对应边界。
     * - 时间复杂度 O(mn)，空间复杂度 O(1)（不计答案）。
     */

import java.util.List;

public class Subsets_78_H {

     * 【子集】
     * 给你整数数组 nums，返回该数组所有可能的子集。
     * 示例：输入 [1,2,3]，输出 [[],[1],[2],[1,2],[3],[1,3],[2,3],[1,2,3]]。
     */
    public List<List<Integer>> subsets(int[] nums) {
        List<List<Integer>> ans = new ArrayList<>();
        backtrack(nums, 0, new ArrayList<>(), ans);
        return ans;
    }

    private void backtrack(int[] nums, int index, List<Integer> path, List<List<Integer>> ans) {
        ans.add(new ArrayList<>(path));
        for (int i = index; i < nums.length; i++) {
            path.add(nums[i]);
            backtrack(nums, i + 1, path, ans);
            path.remove(path.size() - 1);
        }
    }

import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class ThreeSum_15_H {
    /**
     * 【三数之和】
     * 给你整数数组 nums，判断是否存在三元组使和为 0，返回所有不重复三元组。
     * 示例：输入 [-1,0,1,2,-1,-4]，输出 [[-1,-1,2],[-1,0,1]]。
     */
    public List<List<Integer>> threeSum(int[] nums) {
        Arrays.sort(nums);
        List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < nums.length - 2; i++) {
            if (i > 0 && nums[i] == nums[i - 1]) continue;
            int left = i + 1, right = nums.length - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum == 0) {
                    ans.add(Arrays.asList(nums[i], nums[left], nums[right]));
                    while (left < right && nums[left] == nums[left + 1]) left++;
                    while (left < right && nums[right] == nums[right - 1]) right--;
                    left++;
                    right--;
                } else if (sum < 0) left++;
                else right--;
            }
        }
        return ans;
    }

    /**
     * 解题技巧/思路
     * - 排序后固定第一个数，剩余区间用双指针找两数之和。
     * - 跳过相同值避免重复答案。
     * - 时间复杂度 O(n^2)，空间复杂度 O(log n)（排序栈）。
     */

     * 给你整数数组 nums 和整数 k，返回出现频率前 k 高的元素。

        Map<Integer, Integer> count = new HashMap<>();
        for (int num : nums) count.put(num, count.getOrDefault(num, 0) + 1);

        return ans;
